# Specify the file name and extension to search for
$fileToFind = "example.txt"

# Specify the root directory to start the search from (e.g. C:\)
$searchPath = "C:\"

# Search recursively for the file and output the results to the console
try {
    Get-ChildItem $searchPath -Recurse -Filter $fileToFind -ErrorAction Stop | Select-Object FullName
}
catch {
    Write-Host "An error occurred while searching for the file: $($Error[0].Exception.Message)" -ForegroundColor Red
}
